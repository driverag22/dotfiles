    743 #
    744 # Each mouse binding will specify a:
    745 #
    746 # - `mouse`:
    747 #
    748 #   - Middle
    749 #   - Left
    750 #   - Right
    751 #   - Numeric identifier such as `5`

    753 # - `action` (see key bindings)

    755 # And optionally:
    756 #

    760 
    761 # Key bindings
    762 #
    763 # Key bindings are specified as a list of objects. For example, this is the
    764 # default paste binding:
    765 #
    766 # `- { key: V, mods: Control|Shift, action: Paste }`
    767 #
    768 # Each key binding will specify a:
    769 #
    770 # - `key`: Identifier of the key pressed
    771 #
    772 #    - A-Z
    773 #    - F1-F24
    774 #    - Key0-Key9
    775 #
    776 #    A full list with available key codes can be found here:
    777 #    https://docs.rs/glutin/*/glutin/event/enum.VirtualKeyCode.html#variants
    778 #
    779 #    Instead of using the name of the keys, the `key` field also supports using
    780 #    the scancode of the desired key. Scancodes have to be specified as a
    781 #    decimal number. This command will allow you to display the hex scancodes
    782 #    for certain keys:
    783 #
    784 #       `showkey --scancodes`.
    785 #
    786 # Then exactly one of:
    787 #
    788 # - `chars`: Send a byte sequence to the running application
    789 #
    790 #    The `chars` field writes the specified string to the terminal. This makes
    791 #    it possible to pass escape sequences. To find escape codes for bindings
    792 #    like `PageUp` (`"\x1b[5~"`), you can run the command `showkey -a` outside
    793 #    of tmux. Note that applications use terminfo to map escape sequences back
    794 #    to keys. It is therefore required to update the terminfo when changing an
    795 #    escape sequence.
    796 #
    797 # - `action`: Execute a predefined action
    798 #
    799 #   - Copy
    800 #   - Paste
    801 #   - PasteSelection
    802 #   - IncreaseFontSize
    803 #   - DecreaseFontSize
    804 #   - ResetFontSize
    805 #   - ScrollPageUp
    806 #   - ScrollPageDown
    807 #   - ScrollLineUp
    808 #   - ScrollLineDown
    809 #   - ScrollToTop
    810 #   - ScrollToBottom
    811 #   - ClearHistory
    812 #   - Hide
    813 #   - Minimize
    814 #   - Quit
    815 #   - ToggleFullscreen
    816 #   - SpawnNewInstance
    817 #   - ClearLogNotice
    818 #   - ReceiveChar
    819 #   - None
    820 #
    821 #   (macOS only):
    822 #   - ToggleSimpleFullscreen: Enters fullscreen without occupying another space
    823 #
    824 # - `command`: Fork and execute a specified command plus arguments
    825 #
    826 #    The `command` field must be a map containing a `program` string and an
    827 #    `args` array of command line parameter strings. For example:
    828 #       `{ program: "alacritty", args: ["-e", "vttest"] }`
    829 #
    830 # And optionally:
    831 #
    832 # - `mods`: Key modifiers to filter binding actions
    833 #
    834 #    - Command
    835 #    - Control
    836 #    - Option
    837 #    - Super
    838 #    - Shift
    839 #    - Alt
    840 #
    841 #    Multiple `mods` can be combined using `|` like this:
    842 #       `mods: Control|Shift`.
    843 #    Whitespace and capitalization are relevant and must match the example.
    844 #
    845 # - `mode`: Indicate a binding for only specific terminal reported modes
    846 #
    847 #    This is mainly used to send applications the correct escape sequences
    848 #    when in different modes.
    849 #
    850 #    - AppCursor
    851 #    - AppKeypad
    852 #    - Alt
    853 #
    854 #    A `~` operator can be used before a mode to apply the binding whenever
    855 #    the mode is *not* active, e.g. `~Alt`.
    856 #
    857 # Bindings are always filled by default, but will be replaced when a new
    858 # binding with the same triggers is defined. To unset a default binding, it can
    859 # be mapped to the `ReceiveChar` action. Alternatively, you can use `None` for
    860 # a no-op if you do not wish to receive input characters for that binding.
